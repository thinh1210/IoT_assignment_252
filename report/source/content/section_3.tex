\section{Implementation Highlights}

\subsection{System Architecture Overview}
Firmware được tổ chức theo kiến trúc phân lớp kết hợp FreeRTOS task-based runtime. Mục tiêu của cách tổ chức này là tách phần đọc input, điều phối mode, dịch vụ hệ thống và driver phần cứng để giảm coupling, giúp mở rộng thêm TinyML, AP web dashboard và CoreIOT mà không phải dồn toàn bộ logic vào một vòng lặp đơn.

\textbf{Boot sequence của hệ thống}
\begin{enumerate}
    \item Trong \texttt{main.cpp}, hệ thống khởi tạo USB CDC logging và tạo queue trung tâm \texttt{qInputToProcessing}.
    \item \texttt{Main\_FSM::init()} khởi tạo driver output, manual control, config runtime và WifiService facade.
    \item \texttt{InputLayer::init()} khởi tạo button, DHT11, IR receiver, semaphore và các polling task.
    \item \texttt{DisplayService::init()} khởi động OLED, phát welcome animation và tạo task render nền.
    \item \texttt{TaskManager::initModes()} đưa toàn hệ thống vào \texttt{NORMAL\_MODE} ngay sau cold boot.
\end{enumerate}

\subsection{Runtime Task and Event Flow}
\textbf{Luồng dữ liệu cảm biến đến actuator và cloud}
\begin{enumerate}
    \item \texttt{SensorInputService::task\_sensor\_poll} phát \texttt{sensorSemaphore} mỗi \texttt{READ\_INTERVAL = 3000 ms}.
    \item \texttt{InputLayer::task\_manager} nhận semaphore rồi gọi \texttt{SensorInputService::readSensors()} để đọc DHT11.
    \item Nhiệt độ và độ ẩm mới được cập nhật vào runtime state (\texttt{globalTemp}, \texttt{globalHumi}); đồng thời \texttt{PlantCareRuntimeService::updateFromSensor()} chạy TinyML ngay sau khi có mẫu mới.
    \item Sau khi suy luận xong, Input Layer đẩy event \texttt{SENSOR\_DATA\_READY} vào queue \texttt{qInputToProcessing}.
    \item \texttt{Main\_FSM::handleEvent()} nhận event và cập nhật nhánh xử lý môi trường, ví dụ \texttt{NeonController::updateEnvironment()} cho NeoPixel.
    \item Các worker task theo mode dùng cùng nguồn state này để gửi telemetry lên CoreIOT, broadcast ra AP dashboard và hiển thị lên OLED.
\end{enumerate}

\textbf{Vai trò của các mode worker}
\begin{itemize}
    \item \texttt{NORMAL\_MODE}: duy trì WiFi/MQTT connection, gửi telemetry định kỳ mỗi \SI{5}{s}, cho phép TinyML automation điều khiển fan.
    \item \texttt{ACCESSPOINT\_MODE}: phát WiFi AP, broadcast telemetry qua WebSocket mỗi \SI{3}{s}, phục vụ dashboard cấu hình và relay control.
    \item \texttt{MANUAL\_MODE}: vẫn duy trì connectivity/telemetry, nhưng actuator được ưu tiên điều khiển từ IR remote thông qua \texttt{ManualControlService}.
\end{itemize}

\textbf{Display loop độc lập}
\begin{itemize}
    \item \texttt{DisplayService::task\_display\_loop} chạy riêng với chu kỳ \SI{1}{s}.
    \item Task này đọc snapshot nhiệt độ, độ ẩm, trạng thái mode và kết quả TinyML để render trang OLED tương ứng.
    \item Nhờ tách OLED sang task riêng, việc refresh giao diện không chặn luồng đọc sensor hay luồng MQTT/AP.
\end{itemize}

\subsection{Semaphore and Synchronization Design}
\textbf{Input synchronization}
\begin{itemize}
    \item \texttt{btnSemaphore} (counting semaphore) gom sự kiện nút nhấn từ ISR/polling layer.
    \item \texttt{sensorSemaphore} (binary semaphore) đồng bộ chu kỳ đọc cảm biến.
    \item Input manager task lấy semaphore và chuyển thành event cho FSM qua queue \texttt{qInputToProcessing}.
\end{itemize}

\textbf{Driver-level mutual exclusion}
\begin{itemize}
    \item \texttt{LedController::ledMutex}: tránh ghi chồng GPIO LED giữa các task.
    \item \texttt{NeonController::stateMutex}: bảo vệ state môi trường, màu LED, nhịp chớp, manual override.
    \item \texttt{ApService::relayMutex}: bảo vệ relay list và thao tác state relay khi có nhiều luồng WS/automation.
    \item \texttt{ManualControlService::stateMutex}: đồng bộ trạng thái manual IR (fan, pump, green LED, neo LED).
\end{itemize}

\textbf{Logging synchronization}
\begin{itemize}
    \item \texttt{AppLog} có mutex riêng để tránh log bị trộn giữa nhiều task.
    \item Log format thống nhất theo timestamp, level, file, function giúp debug đa task dễ hơn.
\end{itemize}

\subsection{Web Server Redesign in AP Mode}
\begin{itemize}
    \item UI đổi sang dashboard hiện đại (card/gauge/tab), thân thiện hơn bản mẫu cơ bản.
    \item WebSocket xử lý realtime telemetry, relay list, trạng thái TinyML.
    \item HTTP endpoints:
    \begin{itemize}
        \item \texttt{/} trả HTML dashboard
        \item \texttt{/styles.css}
        \item \texttt{/script.js}
    \end{itemize}
    \item WebSocket actions:
    \begin{itemize}
        \item \texttt{add\_relay}
        \item \texttt{toggle\_relay}
        \item \texttt{delete\_relay}
        \item \texttt{save\_settings}
    \end{itemize}
\end{itemize}

\subsection{TinyML Workflow (Data to Device)}
\begin{enumerate}
    \item Thu mẫu nhiệt độ/độ ẩm và gán nhãn qua script thu thập.
    \item Train MLP nhỏ bằng TensorFlow/Keras.
    \item Quantize int8 và export sang \texttt{tflite} + header C.
    \item Nhúng model vào firmware, khởi tạo TFLM interpreter.
    \item Mỗi lần có mẫu DHT mới:
    \begin{itemize}
        \item chuẩn hóa + lượng tử hóa input
        \item \texttt{Invoke()} suy luận
        \item lưu snapshot action/confidence
        \item tự động điều khiển quạt ở NORMAL/AP mode
    \end{itemize}
    \item Publish đồng thời ra OLED, AP dashboard, và CoreIOT telemetry.
\end{enumerate}

\subsection{CoreIOT Workflow and OTA}
\textbf{CoreIOT telemetry flow}
\begin{enumerate}
    \item Kết nối WiFi STA.
    \item Kết nối MQTT tới \texttt{app.coreiot.io:1883}.
    \item Xác thực bằng token cấu hình cho đúng device.
    \item Gửi telemetry định kỳ mỗi \SI{5}{s} ở NORMAL/MANUAL mode.
\end{enumerate}

\textbf{OTA flow}
\begin{enumerate}
    \item Subscribe shared attributes OTA.
    \item Gửi request metadata firmware.
    \item Download và ghi image qua \texttt{Arduino\_ESP32\_Updater}.
    \item Báo trạng thái \texttt{UPDATED} về CoreIOT.
    \item Restart có delay an toàn sau khi update thành công.
\end{enumerate}
