\section{Implementation Highlights}

\subsection{System Architecture Overview}
Chương trình điều khiển được tổ chức theo kiến trúc phân lớp kết hợp với cơ chế thực thi dựa trên các tác vụ của FreeRTOS. Mục tiêu của phương pháp này là tách biệt các phần như đọc dữ liệu đầu vào, điều phối chế độ hoạt động, các dịch vụ hệ thống và trình điều khiển phần cứng. Việc tách biệt giúp giảm thiểu sự phụ thuộc lẫn nhau giữa các thành phần, từ đó tạo điều kiện thuận lợi cho việc mở rộng thêm các tính năng như TinyML, bảng điều khiển web Access Point và CoreIOT mà không cần dồn toàn bộ logic xử lý vào một vòng lặp duy nhất.

\textbf{Boot sequence của hệ thống}
\begin{enumerate}
    \item Trong \texttt{main.cpp}, hệ thống khởi tạo USB CDC logging và tạo queue trung tâm \texttt{qInputToProcessing}.

    \item \texttt{TaskManager::initModes()} đưa toàn hệ thống vào \texttt{NORMAL\_MODE} ngay sau cold boot.
\end{enumerate}

\subsection{Runtime Task and Event Flow}
\textbf{Luồng dữ liệu cảm biến đến actuator và cloud}
\begin{enumerate}
    \item \texttt{SensorInputService::task\_sensor\_poll} phát \texttt{sensorSemaphore} mỗi \texttt{READ\_INTERVAL = 3000 ms}.
    \item \texttt{InputLayer::task\_manager} nhận semaphore rồi gọi \texttt{SensorInputService::readSensors()} để đọc DHT11.
    \item Nhiệt độ và độ ẩm mới được cập nhật vào runtime state (\texttt{globalTemp}, \texttt{globalHumi}); đồng thời \texttt{PlantCareRuntimeService::updateFromSensor()} chạy TinyML ngay sau khi có mẫu mới.
    \item Sau khi suy luận xong, Input Layer đẩy event \texttt{SENSOR\_DATA\_READY} vào queue \texttt{qInputToProcessing}.
    \item \texttt{Main\_FSM::handleEvent()} tiếp nhận sự kiện và cập nhật các nhánh xử lý môi trường, chẳng hạn như \texttt{NeonController::updateEnvironment()} dành cho NeoPixel.
    \item Các tác vụ công việc theo từng chế độ sẽ sử dụng chung nguồn dữ liệu trạng thái này để gửi thông tin giám sát lên CoreIOT, phát dữ liệu ra bảng điều khiển Access Point và hiển thị các thông số lên màn hình OLED.
\end{enumerate}

\textbf{Vai trò của các mode worker}
\begin{itemize}
    \item \texttt{NORMAL\_MODE}: duy trì WiFi/MQTT connection, gửi telemetry định kỳ mỗi \SI{5}{s}, cho phép TinyML automation điều khiển fan.
    \item \texttt{ACCESSPOINT\_MODE}: phát WiFi AP, broadcast telemetry qua WebSocket mỗi \SI{3}{s}, phục vụ dashboard cấu hình và relay control.
    \item \texttt{MANUAL\_MODE}: vẫn duy trì connectivity/telemetry, nhưng actuator được ưu tiên điều khiển từ IR remote thông qua \texttt{ManualControlService}.
\end{itemize}

\textbf{Display loop độc lập}
\begin{itemize}
    \item \texttt{DisplayService::task\_display\_loop} chạy riêng với chu kỳ \SI{1}{s}.
    \item Task này đọc snapshot nhiệt độ, độ ẩm, trạng thái mode và kết quả TinyML để render trang OLED tương ứng.
    \item Nhờ tách OLED sang task riêng, việc refresh giao diện không chặn luồng đọc sensor hay luồng MQTT/AP.
\end{itemize}

\subsection{Semaphore and Synchronization Design}
Hệ thống dùng semaphore và mutex theo ba vai trò khác nhau: đồng bộ producer-consumer ở lớp input, bảo vệ state hoặc phần cứng dùng chung ở lớp driver/service, và tuần tự hóa đầu ra log. Cách chia này giúp các task sinh dữ liệu chạy ngắn gọn, còn xử lý nghiệp vụ tập trung ở FSM thông qua queue sự kiện.

\textbf{Input synchronization}
\begin{itemize}
    \item Trong source hiện tại, sự kiện nút nhấn được phát từ \textit{polling task}, chưa dùng ISR. Hàm \texttt{InputLayer::init()} tạo \texttt{btnSemaphore = xSemaphoreCreateCounting(20, 0)} để gom nhiều sự kiện nút liên tiếp mà không bị mất nếu consumer xử lý chậm hơn trong một khoảng ngắn.
    \item Task \texttt{ButtonInputService::task\_button\_poll()} chạy chu kỳ 50\,ms, gọi \texttt{ButtonHandler::loop()} để debounce, nhận diện \texttt{SINGLE\_CLICK}, \texttt{DOUBLE\_CLICK}, \texttt{TRIPLE\_CLICK} và \texttt{LONG\_PRESS}. Khi phát hiện event hợp lệ, lớp này chỉ cần \texttt{xSemaphoreGive(btnSemaphore)} thay vì truy cập trực tiếp FSM.
    \item Task trung gian \texttt{InputLayer::task\_manager()} lấy \texttt{btnSemaphore} với timeout ngắn 10\,ms. Khi lấy được semaphore, task này gọi \texttt{ButtonInputService::handleButtonEvents()} để chuyển event vật lý thành \texttt{SystemEvent} và đưa vào queue xử lý chung \texttt{qProcessing}.
    \item \texttt{sensorSemaphore} được tạo bằng \texttt{xSemaphoreCreateBinary()}. Với cảm biến, hệ thống chỉ cần biết ``đã tới chu kỳ đọc mới'' nên binary semaphore là phù hợp hơn counting semaphore.
    \item Task \texttt{SensorInputService::task\_sensor\_poll()} chỉ có nhiệm vụ chờ \texttt{READ\_INTERVAL} rồi \texttt{xSemaphoreGive(sensorSemaphore)}. Sau đó \texttt{InputLayer::task\_manager()} nhận tín hiệu này, gọi \texttt{readSensors()}, cập nhật dữ liệu DHT và đẩy event \texttt{SENSOR\_DATA\_READY} cho FSM.
\end{itemize}

\begin{lstlisting}[language=C++,caption={Counting semaphore cho luong nut nhan}]
btnSemaphore = xSemaphoreCreateCounting(20, 0);
btnHandler->setSemaphore(btnSemaphore);

// Producer: polling + debounce layer
if (btn.clickCount >= 3) {
  btn.lastEvent = ButtonEvent::TRIPLE_CLICK;
  xSemaphoreGive(_eventSemaphore);
}

// Consumer: input manager
if (xSemaphoreTake(btnSemaphore, pdMS_TO_TICKS(10)) == pdPASS) {
  ButtonInputService::handleButtonEvents(btnHandler, qProcessing);
}
\end{lstlisting}

\begin{lstlisting}[language=C++,caption={Binary semaphore cho chu ky doc cam bien}]
void SensorInputService::task_sensor_poll(void *param) {
  SemaphoreHandle_t syncSem = (SemaphoreHandle_t)param;
  while (1) {
    vTaskDelay(pdMS_TO_TICKS(READ_INTERVAL));
    xSemaphoreGive(syncSem);
  }
}

if (xSemaphoreTake(sensorSemaphore, pdMS_TO_TICKS(10)) == pdPASS) {
  SensorInputService::readSensors(qProcessing);
}
\end{lstlisting}

\textbf{Driver-level mutual exclusion}
\begin{itemize}
    \item \texttt{LedController::ledMutex} bảo vệ GPIO LED xanh và biến \texttt{ledState}. Mọi thao tác \texttt{setState()}, \texttt{toggle()} hay \texttt{blink()} đều đi qua cùng một mutex, nên nhiều task không thể ghi chồng lên cùng chân GPIO trong cùng thời điểm.
    \item \texttt{NeonController::stateMutex} bảo vệ toàn bộ state nội bộ của NeoPixel: màu môi trường, chu kỳ chớp, cờ \texttt{manualOverride}, trạng thái sáng/tắt trong manual mode. Điểm quan trọng là blink task chỉ sao chép snapshot state trong vùng khóa rồi nhả mutex trước khi gọi \texttt{showColor()} và \texttt{vTaskDelay()}, nhờ đó critical section ngắn và ít chặn các task khác.
    \item \texttt{ApService::relayMutex} bảo vệ \texttt{relayList} khi có nhiều nguồn cập nhật như WebSocket dashboard, automation và manual control. Các hàm \texttt{setRelayState()}, \texttt{getRelayState()}, \texttt{syncAutomationRelays()} và các nhánh \texttt{add/toggle/delete relay} cùng dùng mutex này với timeout 200\,ms để giữ cho trạng thái trên RAM, GPIO, NVS và dashboard nhất quán.
    \item \texttt{ManualControlService::stateMutex} đồng bộ snapshot manual gồm \texttt{fanOn}, \texttt{pumpOn}, \texttt{greenLedOn}, \texttt{neoLedOn} và cờ \texttt{active}. Cài đặt hiện tại chỉ giữ mutex đủ lâu để cập nhật state nội bộ, sau đó nhả khóa rồi mới gọi sang \texttt{ApService}, \texttt{LedController} hoặc \texttt{NeonController}. Mẫu này giúp tránh giữ lock quá lâu và giảm nguy cơ lock lồng nhau.
\end{itemize}

\begin{lstlisting}[language=C++,caption={Mau cap nhat state ngan gon roi moi goi driver}]
if (xSemaphoreTake(stateMutex, pdMS_TO_TICKS(50)) != pdPASS) {
  return;
}

state.neoLedOn = !state.neoLedOn;
const bool nextState = state.neoLedOn;
xSemaphoreGive(stateMutex);

NeonController::setManualOverride(true, nextState);
\end{lstlisting}

\textbf{Logging synchronization}
\begin{itemize}
    \item \texttt{AppLog} tạo mutex riêng \texttt{s\_logMutex} để mọi task cùng ghi log mà không bị xen dòng.
    \item Mỗi message được format hoàn chỉnh vào buffer trước, sau đó mới lấy mutex khi ghi ra \texttt{Serial}. Nhờ vậy thời gian giữ lock ngắn hơn so với cách vừa format vừa ghi trực tiếp.
    \item Mẫu log thống nhất theo \texttt{[timestamp][level][file:line] function(): [tag] message}, rất hữu ích khi cần truy vết tương tác giữa nhiều task FreeRTOS.
\end{itemize}

\begin{lstlisting}[language=C++,caption={Mutex cho logging da task}]
snprintf(lineBuffer, sizeof(lineBuffer),
         "[%6lu][%c][%s:%u] %s(): [%s] %s\r\n",
         timestamp, level, file, line, function, tag, message);

if (xSemaphoreTake(s_logMutex, kLogMutexWait) == pdPASS) {
  writeToSerialLocked(lineBuffer, strlen(lineBuffer));
  xSemaphoreGive(s_logMutex);
}
\end{lstlisting}

\subsection{TinyML Workflow (Data to Device)}
\begin{enumerate}
    \item Thu mẫu nhiệt độ/độ ẩm và gán nhãn qua script thu thập.
    \item Train MLP nhỏ bằng TensorFlow/Keras.
    \item Quantize int8 và export sang \texttt{tflite} + header C.
    \item Nhúng model vào firmware, khởi tạo TFLM interpreter.
    \item Mỗi lần có mẫu DHT mới:
    \begin{itemize}
        \item chuẩn hóa + lượng tử hóa input
        \item \texttt{Invoke()} suy luận
        \item lưu snapshot action/confidence
        \item tự động điều khiển quạt ở NORMAL/AP mode
    \end{itemize}
    \item Publish đồng thời ra OLED, AP dashboard, và CoreIOT telemetry.
\end{enumerate}

\subsection{CoreIOT Workflow and OTA}
\textbf{CoreIOT telemetry flow}
\begin{enumerate}
    \item Kết nối WiFi STA.
    \item Kết nối MQTT tới \texttt{app.coreiot.io:1883}.
    \item Xác thực bằng token cấu hình cho đúng device.
    \item Gửi telemetry định kỳ mỗi \SI{5}{s} ở NORMAL/MANUAL mode.
\end{enumerate}

\textbf{OTA flow}
\begin{enumerate}
    \item Subscribe shared attributes OTA.
    \item Gửi request metadata firmware.
    \item Download và ghi image qua \texttt{Arduino\_ESP32\_Updater}.
    \item Báo trạng thái \texttt{UPDATED} về CoreIOT.
    \item Restart có delay an toàn sau khi update thành công.
\end{enumerate}
